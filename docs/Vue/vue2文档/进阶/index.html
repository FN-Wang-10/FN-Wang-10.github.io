处理边界​

访问根实例($root,$parent)​

可以通过 this.$root的方式，在任意组件获取根实例
可以通过this.$parent 获取父组件的实例

可以获取到状态和方法


<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <div id="app">
        <child></child>
    </div>

    <script src="../../Vue.js"></script>
    <script>
        const Child2 = {
            template: `
                <div>child2组件{{ $root.count }}-{{ $parent.msg }}</div>
            `,
            mounted() {
                // 可以通过this.$root的方式，在任意组件获取根实例
                console.log(this.$root);
                console.log(this.$root.count);
                this.$root.fn();
                // 可以通过this.$parent获取父组件的实例
                console.log(this.$parent);
                console.log(this.$parent.msg);
            }
        };
        const Child = {
            data() {
                return {
                    msg: 20,
                };
            },
            template: `
                <div>child组件-<child2></child2></div>
            `,
            components: {
                Child2,
            },
        };
        new Vue({
            el: '#app',
            data: {
                count: 10,
            },
            components: {
                Child,
            },
            mounted() {
                console.log(this);
            },
            methods: {
                fn() {
                    console.log(123);
                }
            }
        });
    </script>
</body>

</html>

获取子组件实例，获取 dom($refs)​

可以在父组件的子组件标签上面添加 ref 属性,自定义一个名字
在父组件的 mounted 里面通过 this.$refs.自定义的名字，获取到子组件的实例
这种方式只能在页面渲染完毕（mounted 里面）才能拿到
这种方式仅可以获取，不能修改子组件里面的数据

this.$refs不仅可以用于获取子组件实例，还可以去获取dom元素
获取dom元素的方式是，在元素上添加ref属性，自定义一个名字
在父组件的mounted里面通过this.$refs.自定义的名字，获取到这个 dom 元素
这种方式只能在页面渲染完毕（mounted 里面）才能拿到

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <div id="app">
        <!-- 父组件获取子组件的实例 -->
        <!-- 可以在父组件的子组件标签上面添加ref属性,自定义一个名字 -->
        <!-- 在父组件的mounted里面通过this.$refs.自定义的名字，获取到子组件的实例 -->
        <!-- 这种方式只能在页面渲染完毕（mounted里面）才能拿到 -->
        <!-- 这种方式仅可以获取，不能修改子组件里面的数据 -->
        <child ref="abc"></child>

        <br>
        <!-- this.$refs不仅可以用于获取子组件实例，还可以去获取dom元素 -->
        <!-- 获取dom元素的方式是，在元素上添加ref属性，自定义一个名字 -->
        <!-- 在父组件的mounted里面通过this.$refs.自定义的名字，获取到这个dom元素 -->
        <!-- 这种方式只能在页面渲染完毕（mounted里面）才能拿到 -->
        <input type="text" ref="ipt">
    </div>

    <script src="../../Vue.js"></script>
    <script>
        const Child = {
            template: `
                <div>child组件</div>
            `,
            data() {
                return {
                    msg: 'hello',
                };
            },
        }
        new Vue({
            el: '#app',
            data: {},
            components: {
                Child,
            },
            mounted() {
                console.log(this.$refs.abc);
                console.log(this.$refs.abc.msg);
                // console.log(document.querySelector('input'))//原生js的方式基本不用
                console.log(this.$refs.ipt);
            }
        });
    </script>
</body>

</html>

依赖注入(provide,inject)​

在父组件中使用 provide 选项，返回一个对象，在子孙组件使用 inject 选项接收（数组） 依赖（provide 选项），向孙组件传递内容
穿透式传递
写法和 data 选项一样，函数返回对象
它是没有响应式的(vue2)，vue3 有了响应式

注入(inject)，可以获取父元素传来的内容,使用数组的方式接收

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <div id="app">
        <child1></child1>
    </div>

    <script src="../../Vue.js"></script>
    <script>
        const Child2 = {
            template: `
                <div>child2组件-{{ count }}</div>
            `,
            //注入(inject)，可以获取父元素传来的内容
            inject: ["count"],
        };
        const Child1 = {
            template: `
                 <div>child1组件 - <child2></child2></div>
            `,
            components: {
                Child2,
            }
        };
        new Vue({
            el: '#app',
            data: {
                count: 20,
            },
            components: {
                Child1,
            },
            // 依赖（provide选项），向孙组件传递内容
            // 穿透式传递
            // 写法和data选项一样，函数返回对象
            // 它是没有响应式的(vue2)，vue3有了响应式

            // 注入(inject)，可以获取父元素传来的内容,使用数组的方式接收
            //
            provide() {
                return {
                    count: this.count,
                };
            },
        });
    </script>
</body>

</html>

强制更新($forceUpdate() $set())​

响应式失效的三种方法

  1. 直接修改数组长度this.list.length = 2
  2. 通过数组下标修改this.list[2] = 10
  3. 为对象添加新属性this.obj.c = 20

解决 [...this.list]
[].concat(this.list)
this.list.splice(2)

{...this.obj}
Object.assign({}, this.obj)

强制更新
this.$forceUpdate() //可以更新，但是尽量少用 this.$set(this.obj, 'c', 100)

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <div id="app">
        <h3>强制更新</h3>
        <p>{{ list }}</p>
        <p>{{ obj }}</p>
        <button @click="handleClick">点击</button>
    </div>

    <script src="../../Vue.js"></script>
    <script>
        new Vue({
            el: '#app',
            data: {
                list: [1, 2, 3],
                obj: {
                    a: 2,
                    b: 3,
                }
            },
            methods: {
                handleClick() {
                    // 不能强制更新的三种情况
                    // this.list.length = 2; //不可以更新
                    // this.list[2] = 10; // 不可以更新
                    // this.obj.a = 100;  //可以更新
                    // this.obj.c = 20; // 不可以更新

                    // 数组
                    // this.list = [...this.list]; //数组方法一
                    // this.list = [].concat(this.list); //数组方法一
                    // this.list.splice(2); //数组方法三
                    // 对象
                    // this.obj = {...this.obj}; //对象方法一
                    // this.obj = Object.assign({}, this.obj); //对象方法二

                    // 强制更新
                    // this.$forceUpdate(); //可以更新，但是尽量少用
                    // this.$set(this.obj, 'c', 100);
                },
            },
        });
    </script>
</body>

</html>

过渡和动画(transition)​

过渡(transition)动画(animation)
过渡需要用户是触发，动画不需要用户触发
过渡只能做一次，动画可以一直循环
过度无法控制中间的过程，动画可以通过关键帧控制，更精细
有中间值的属性可以参与过渡

混入(mixins 选项[])​

混入： 分发可复用功能

创建混入对象
跟组件创建的方式是一样，它没有 template 选项

如果说 key 不同会进行合并，key 相同会替换，以组件自身的 key 值为准
生命周期是直接合并，不会替换的

使用混入对象,数组的形式接收
props,inject 选项对象是需要加引号的
mixins 选项对象不能加引号的

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <div id="app">
        <h3>混入-{{ count }}</h3>
        <child></child>
    </div>

    <script src="../../Vue.js"></script>
    <script>
        // 混入： 分发可复用功能

        // 创建混入对象
        // 跟组件创建的方式是一样，它没有template选项

        // 如果说key不同会进行合并，key相同会替换，以组件自身的key值为准
        // 生命周期是直接合并，不会替换的
        const myMixin = {
            data() {
                return {
                    count: 10,
                };
            },
            methods: {
                fn() {
                    console.log('fn');
                },
            },
            mounted() {
                console.log(123);
            },
        };
        const myMixin2 = {
            data() {
                return {
                    num: 200,
                };
            },
        }
        const Child = {
            // 使用混入对象,数组的形式接收
            // props,inject选项对象是需要加引号的
            // mixins选项对象不能加引号的
            mixins: [myMixin, myMixin2],
            data() {
                return {
                    // count: 10,
                    msg: 100,
                    count: 20, //key相同会替换，以组件自身的key值为准
                };
            },
            template: `
                <div>child组件-{{ count }}-{{ msg }}-{{ num }}</div>
            `,
            mounted() {
                this.fn();

            },
        }

        new Vue({
            el: '#app',
            data: {
                // count: 10,
            },
            mixins: [myMixin],
            components: {
                Child,
            },
            mounted() {
                this.fn();
            }
        });
    </script>
</body>

</html>

                //     // 五个钩子函数，相当于自定义指令的五个生命周期
                //     // 钩子函数可以接收四个参数
                //     // el: 表示挂载的元素
                //     // binding: 表示一个很多参数组成的对象（对象有key和value）
                //     // vnode: 表示新的虚拟DOM
                //     // oldVnode: 表示旧的虚拟DOM
                //     bind(el, binding, vnode, oldVnode) {},
                //     inserted(el, binding, vnode, oldVnode) {},
                //     update(el, binding, vnode, oldVnode) {},
                //     componentUpdated(el, binding, vnode, oldVnode) {},
                //     unbind(el, binding, vnode, oldVnode) {},
                // },
                // 在很多时候，你可能想在 bind 和 update 时触发相同行为，而不关心其它的钩子。比如这样写：
                abc(el, binding, vnode, oldVnode) {
                    console.log(el); // el表示挂载的元素
                    console.log(binding); // binding表示一个很多参数组成的对象
                    //     //    arg: "style" 指令的参数
                    //     //    def(对象): bind和updata  表示钩子函数
                    //     //    expression: "900"
                    //     //    modifiers（对象）: 30和red  表示修饰符
                    //     //    name: "abc" 指令名称
                    //     //    rawName: "v-abc:style.30.red" 指令名称
                    //     //    value: 900
                    // 解构对象
                    const {
                        arg,
                        modifiers,
                        value
                    } = binding;
                    const [size, color] = Object.keys(modifiers);
                    el[arg].fontSize = size + "px";
                    el[arg].color = color;
                    el[arg].fontWeight = value;
                },
            },
        });
    </script>
</body>

过滤器(filters 选项)​

什么时候用过滤器？
文本格式化

双大括号和 v-bind 都可以使用过滤器
过滤器可以串联

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body>
    <div id="app">
      <!-- 文本格式化 -->
      <!-- 双大括号和v-bind都可以使用过滤器 -->
      <!-- 过滤器可以串联 -->
      <!-- <p>{{ sex === 1 ? "男" : "女" }}</p> -->
      <p>{{ sex | formatSex | formatSex2 }}</p>
      <p :title="sex | formatSex">{{ time | formatTime }}</p>
    </div>

    <script src="../../Vue.js"></script>
    <script>
      new Vue({
        el: "#app",
        data: {
          sex: 1,
          time: 1649216924644,
        },
        filters: {
          formatSex(val) {
            return val === 1 ? "男" : "女";
          },
          formatSex2(val) {
            return val === "男" ? "♂" : "♀";
          },
          formatTime(val) {
            var date = new Date(val);
            var YY = date.getFullYear() + "-";
            var MM =
              (date.getMonth() + 1 < 10
                ? "0" + (date.getMonth() + 1)
                : date.getMonth() + 1) + "-";
            var DD =
              date.getDate() < 10 ? "0" + date.getDate() : date.getDate();
            var hh =
              (date.getHours() < 10 ? "0" + date.getHours() : date.getHours()) +
              ":";
            var mm =
              (date.getMinutes() < 10
                ? "0" + date.getMinutes()
                : date.getMinutes()) + ":";
            var ss =
              date.getSeconds() < 10
                ? "0" + date.getSeconds()
                : date.getSeconds();
            return YY + MM + DD + " " + hh + mm + ss;
          },
        },
      });
    </script>
  </body>
</html>
